268 Result<Unit, OutOfBounds> copy_to(Slice<std::remove_const_t<T>> dest) const noexcept {
269 if (dest.size() < m_size) {
270 return Err(OutOfBounds{m_size, dest.size()});
271 }
272 if constexpr (std::is_trivially_copyable_v<T>) {
273 std::memmove(dest.data(), m_data, m_size * sizeof(T));
274 } else {
275 for (size_type i = 0; i < m_size; ++i) {
276 dest[i] = m_data[i];
277 }
278 }
